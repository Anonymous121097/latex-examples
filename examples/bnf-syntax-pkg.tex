\setcounter{tcb@cnt@example}{3}
\begin{example}[]{\texttt{syntax} pkg, default settings}
  % \usepackage{syntax}
  
  \begin{grammar}
    <statement> ::= <ident> `=' <expr> 
    \alt `for' <ident> `=' <expr> `to' <expr> `do' <statement> 
    \alt `{' <stat-list> `}' 
    \alt <empty> 
    
    <stat-list> ::= <statement> `;' <stat-list> | <statement> 
  \end{grammar}
\end{example}
