\usepackage{expl3}
\usepackage{fontspec}

\ExplSyntaxOn
\makeatletter

\newfontfamily\@cciconsFamily{cc-icons}

\msg_set:nnn { ccicons } { invalid~name }
  {
    CC~Icon~with~name~``#1''~does~not~exist.~Use~``cc-cc''~instead.
  }

  {
    \group_begin:
    \@cciconsFamily
    %% ref: https://github.com/cc-icons/cc-icons/blob/master/css/cc-icons.css
    \str_case:nnF {#1}
      {
        %% round icons
        {cc-sa}           {a}
        {cc-by}           {b}
        {cc-cc}           {c}
        {cc-nd}           {d}
        {cc-nc-eu}        {e}

        {cc-nc}           {n}
        {cc-zero}         {o}
        {cc-pd}           {p}
        {cc-pd-alt}       {q}
        {cc-remix}        {r}
        {cc-share}        {s}
        %% license banners
        {cc-srr}          {A}
        {cc-BY}           {B}
        {cc-logo}         {C}
        {cc-by-nd}        {D}
        {cc-by-nc-eu}     {E}
        {cc-by-nc-nd-eu}  {F}
        {cc-by-nc-sa-eu}  {G}
        {cc-by-nc-jp}     {J}

  }
