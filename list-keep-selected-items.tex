\documentclass{article}
\usepackage{xparse}

\makeatletter
\let\enumerate@backup\enumerate
\let\endenumerate@backup\endenumerate

\ExplSyntaxOn
\tl_new:N \l_enumerate_body_tl

\tl_new:N  \l_item_temp_tl
\seq_new:N \l_list_body_seq
\seq_new:N \l_list_new_body_seq
\seq_new:N \l_list_keep_items_seq
\int_new:N \l_items_count_int

\RenewDocumentEnvironment{enumerate}{ o b }{
  \int_set:Nn \l_items_count_int {0}
  \seq_clear:N \l_list_new_body_seq
  
  \seq_set_split:Nnn \l_list_body_seq {\item} {#2}
  
  \seq_pop_left:NN \l_list_body_seq \l_item_temp_tl
  \tl_if_blank:VF \l_item_temp_tl {ERROR}
  
  \seq_map_inline:Nn \l_list_body_seq
    {
      \int_incr:N \l_items_count_int
      \seq_pop_left:NN \l_list_body_seq \l_item_temp_tl
      % O(mn) time complexity. Can be O(n).
      \seq_if_in:NVTF \l_list_keep_items_seq \l_items_count_int
        {
          \seq_put_right:Nn \l_list_new_body_seq {\item}
          \seq_put_right:NV \l_list_new_body_seq \l_item_temp_tl
        }
        {
          \seq_put_right:Nn \l_list_new_body_seq { \stepcounter{enumi} }
        }
    }
}{
  \tl_if_novalue:nTF {#1}
    { \begin{enumerate@backup} }
    { \begin{enumerate@backup}[#1] }
    \seq_use:Nn \l_list_new_body_seq {}
  \end{enumerate@backup}
}

\NewDocumentCommand{\keepItems}{m}{
  \seq_set_split:Nnn \l_list_keep_items_seq {,} {#1}
}
\ExplSyntaxOff
\makeatother

\begin{document}
